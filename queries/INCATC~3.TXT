INSERT INTO FACCSSDB.INCTHDTA
SELECT 	DISTINCT 
	I.INCIDENT_CD AS INCIDENT_NUM, 
	 (SELECT TIMESTAMPDIFF (2, CAST((TIMESTAMP(DPTSCN_DATE, DPTSCN_TIME)) - 
		       TIMESTAMP(ARVDST_DATE, ARVDST_TIME) AS CHAR(22))) 
	 FROM FACCSDB.INCIDENT_UNITS1 U
	 WHERE U.INCIDENT_CD = I.INCIDENT_CD) AS Travel_Time

FROM	FACCSDB.INCIDENTS1 I,
	FACCSDB.INCIDENT_UNITS1 U,
	FACCSDB.PATIENTS1 P,
	FACCSDB.DIAGCODES D

WHERE	I.LOC_HOSP_CATCH_CD = 1 AND
	I.CREATE_DATE BETWEEN DATE('01/31/1999') AND DATE('06/01/1999') AND
	I.APCTRANSPORT = 'R' AND
	P.DIAG_CD IN D.DIAG_CD	AND
	I.INCIDENT_CD = P.INCIDENT_CD
