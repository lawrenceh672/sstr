CREATE VIEW TOTEMS (INCIDENT_CD, TIME) AS ( SELECT	P.INCIDENT_CD, TIMESTAMPDIFF (2, CAST((TIMESTAMP(DISP_DATE, DISP_TIME)) - TIMESTAMP(ARVDST_DATE, ARVDST_TIME) AS CHAR(22))) AS TRAVTIME FROM	FACCSDB.INCIDENTS1 I, FACCSDB.INCIDENT_UNITS1 U, FACCSDB.PATIENTS1 P, FACCSDB.DIAGCODES D WHERE	P.DIAG_CD IN D.DIAG_CD	AND P.INCIDENT_CD = I.INCIDENT_CD	AND U.INCIDENT_CD = I.INCIDENT_CD	AND U.DISP_DATE BETWEEN DATE('01/31/1999') AND CURRENT DATE )
